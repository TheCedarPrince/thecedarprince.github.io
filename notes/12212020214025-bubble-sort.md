```julia
function bubblesort(arr)
    size = length(arr)
    while size > 1
        for curr = 2:size
            prior = curr - 1
            if arr[curr] < arr[prior]
                arr[curr], arr[prior] = arr[prior], arr[curr]
            end
        end
        size -= 1
    end
    return arr
end
```
